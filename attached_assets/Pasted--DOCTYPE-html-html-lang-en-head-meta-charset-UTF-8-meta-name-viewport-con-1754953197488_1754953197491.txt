        function openModal(id, icon, title, description) {
            document.getElementById('modalIcon').textContent = icon;
            document.getElementById('modalTitle').textContent = title;
            document.getElementById('modalDescription').textContent = description;
            document.getElementById('achievementModal').style.display = 'flex';

        function closeModal() {
            document.getElementById('achievementModal').style.display = 'none';
        }

        function filterBadges(filterType) {
            const achievementsGrid = document.querySelector('.achievements-grid');
            const allCards = Array.from(achievementsGrid.querySelectorAll('.achievement-card'));
            const filterButtons = document.querySelectorAll('.filter-btn');
            
            // Remove active class from all buttons
            filterButtons.forEach(btn => btn.classList.remove('active'));
            
            // Add active class to clicked button
            document.querySelector(`[data-filter="${filterType}"]`).classList.add('active');
            
            // Reset all cards to full opacity first
            allCards.forEach(card => {
                card.style.opacity = '1';
            });
            
            let filteredCards = [];
            
            switch(filterType) {
                case 'completed':
                    filteredCards = allCards.filter(card => card.classList.contains('achieved'));
                    break;
                case 'in-progress':
                    filteredCards = allCards.filter(card => !card.classList.contains('achieved'));
                    break;
                case 'hall-of-fame':
                    filteredCards = allCards.filter(card => card.classList.contains('hall-of-fame'));
                    break;
                case 'superstar':
                    filteredCards = allCards.filter(card => card.classList.contains('superstar'));
                    break;
                case 'all-star':
                    filteredCards = allCards.filter(card => card.classList.contains('all-star'));
                    break;
                case 'starter':
                    filteredCards = allCards.filter(card => card.classList.contains('starter'));
                    break;
                case 'prospect':
                    filteredCards = allCards.filter(card => card.classList.contains('prospect'));
                    break;
                case 'all':
                default:
                    filteredCards = allCards;
                    break;
            }
            
            // Clear the grid
            achievementsGrid.innerHTML = '';
            
            // For tier-specific filters, sort by completion status (completed first)
            if (['hall-of-fame', 'superstar', 'all-star', 'starter', 'prospect'].includes(filterType)) {
                const completedCards = filteredCards.filter(card => card.classList.contains('achieved'));
                const inProgressCards = filteredCards.filter(card => !card.classList.contains('achieved'));
                
                // Add completed cards first
                completedCards.forEach(card => {
                    card.style.opacity = '1';
                    achievementsGrid.appendChild(card);
                });
                
                // Add in-progress cards second
                inProgressCards.forEach(card => {
                    card.style.opacity = '1';
                    achievementsGrid.appendChild(card);
                });
                
                // Add remaining cards with reduced opacity
                const remainingCards = allCards.filter(card => !filteredCards.includes(card));
                remainingCards.forEach(card => {
                    card.style.opacity = '0.3';
                    achievementsGrid.appendChild(card);
                });
            } else {
                // For other filters, add filtered cards first (full opacity)
                filteredCards.forEach(card => {
                    card.style.opacity = '1';
                    achievementsGrid.appendChild(card);
                });
                
                // Add remaining cards with reduced opacity (except for 'all' filter)
                if (filterType !== 'all') {
                    const remainingCards = allCards.filter(card => !filteredCards.includes(card));
                    remainingCards.forEach(card => {
                        card.style.opacity = '0.3';
                        achievementsGrid.appendChild(card);
                    });
                }
            }
        }

        });
