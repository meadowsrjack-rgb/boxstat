3. Update AppDelegate.swift (The Token Handler)

In Replit, open ios/App/AppDelegate.swift. Replace the entire file with this.

Why: This removes all Firebase code and replaces it with the code that prints the Apple Device Token directly to your console.

Swift
import UIKit
import UserNotifications

@main
class AppDelegate: UIResponder, UIApplicationDelegate, UNUserNotificationCenterDelegate {

    func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
        
        // 1. Set up the notification delegate
        UNUserNotificationCenter.current().delegate = self
        
        // 2. Request permission immediately on launch
        let authOptions: UNAuthorizationOptions = [.alert, .badge, .sound]
        UNUserNotificationCenter.current().requestAuthorization(options: authOptions) { granted, error in
            print("ðŸ”” Notification Permission Granted: \(granted)")
            if let error = error {
                print("âŒ Error: \(error)")
            }
        }
        
        // 3. Register with Apple Push Notification Service (APNs)
        application.registerForRemoteNotifications()
        
        return true
    }

    // --- SUCCESS: We got a token from Apple ---
    func application(_ application: UIApplication, didRegisterForRemoteNotificationsWithDeviceToken deviceToken: Data) {
        let tokenParts = deviceToken.map { data in String(format: "%02.2hhx", data) }
        let token = tokenParts.joined()
        print("âœ… DEVICE TOKEN: \(token)")
        // This is the token you will eventually need to copy into your dashboard
    }

    // --- FAILURE: Something went wrong ---
    func application(_ application: UIApplication, didFailToRegisterForRemoteNotificationsWithError error: Error) {
        print("âŒ Failed to register for notifications: \(error)")
    }

    // --- MARK: UISceneSession Lifecycle (Required for modern iOS) ---
    func application(_ application: UIApplication, configurationForConnecting connectingSceneSession: UISceneSession, options: UIScene.ConnectionOptions) -> UISceneConfiguration {
        return UISceneConfiguration(name: "Default Configuration", sessionRole: connectingSceneSession.role)
    }

    func application(_ application: UIApplication, didDiscardSceneSessions sceneSessions: Set<UISceneSession>) {}
}
