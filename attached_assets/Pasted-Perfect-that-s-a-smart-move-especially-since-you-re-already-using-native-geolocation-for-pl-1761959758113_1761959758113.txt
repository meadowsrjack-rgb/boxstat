## ğŸ§© Step 3 â€“ Replace address search with Nominatim fetch

```tsx
import { useState } from "react";

export function LocationSearch({ onSelect }: { onSelect: (loc: any) => void }) {
  const [query, setQuery] = useState("");
  const [results, setResults] = useState<any[]>([]);

  const search = async (value: string) => {
    setQuery(value);
    if (value.length < 3) return setResults([]);
    const res = await fetch(
      `https://nominatim.openstreetmap.org/search?format=json&q=${encodeURIComponent(value)}&addressdetails=1&limit=5`
    );
    const data = await res.json();
    setResults(data);
  };

  return (
    <div className="space-y-2">
      <input
        className="w-full rounded-md border p-2"
        placeholder="Search location"
        value={query}
        onChange={(e) => search(e.target.value)}
      />
      <ul>
        {results.map((r) => (
          <li
            key={r.place_id}
            className="cursor-pointer hover:bg-gray-100 p-1 rounded"
            onClick={() => {
              setQuery(r.display_name);
              setResults([]);
              onSelect({
                name: r.display_name,
                lat: parseFloat(r.lat),
                lng: parseFloat(r.lon),
              });
            }}
          >
            {r.display_name}
          </li>
        ))}
      </ul>
    </div>
  );
}
```

## ğŸ§© Step 4 â€“ Display map with Leaflet

```tsx
import { MapContainer, TileLayer, Marker, Popup } from "react-leaflet";

export function MapPreview({ lat, lng }: { lat: number; lng: number }) {
  if (!lat || !lng)
    return <p className="text-muted-foreground">Location not specified.</p>;

  return (
    <MapContainer
      center={[lat, lng]}
      zoom={15}
      scrollWheelZoom={false}
      className="h-64 w-full rounded-lg"
    >
      <TileLayer
        url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
        attribution="&copy; OpenStreetMap contributors"
      />
      <Marker position={[lat, lng]}>
        <Popup>Event location</Popup>
      </Marker>
    </MapContainer>
  );
}
```

This uses the free OpenStreetMap tile server; you can swap to your own or Mapbox later if needed.
